def getGrade(Number number) {
    def result
    switch (number) {
        case 0..<60:
            result = '不及格'
            break
        case 60..<70:
            result = '及格'
            break
        case 70..<80:
            result = '良好'
            break
        case 80..100:
            result = '优秀'
            break
        default:
            result = "error!"
    }
    return result
}
//良好

eg:

class Person {
    String name
    Integer age
}


class Entry {
    static void main(def args) {
        println 'start up '
        ApplicationManager.init()
        println 'init finish'

        def person = PersonManager.createPerson('jerry',24)
        println "name is ${person.name},age is ${person.age}"
    }
}
class ApplicationManager {
    static void init() {
        ExpandoMetaClass.enableGlobally()
        //为第三方类添加方法
        Person.metaClass.static.createPerson = { String name, int age -> new Person(name: name, age: age)
        }
    }
}


class PersonManager {
    static Person createPerson(String name, int age) {
        return Person.createPerson(name,age)
    }
}
